```typescript
const __dirname = dirname(fileURLToPath(import.meta.url))
const MIGRATIONS_DIR = resolve(__dirname, "../../../../packages/database/migrations")
const SEED_SQL = resolve(MIGRATIONS_DIR, "0001_init_seed.sql")

  const db = drizzle(sqlite, { schema })

  return { db, sqlite }
```

```typescript
// apps/dashboard/src/proxy.test.ts
import { describe, it, expect, vi, beforeEach } from "vitest"
import { NextRequest } from "next/server"

vi.mock("@/lib/auth")

import { getAuth } from "@/lib/auth"
import { proxy } from "./proxy"

const req = (pathname: string) =>
  new NextRequest(new URL(`http://localhost${pathname}`))

const mockSession = (user: { id: string } | null) =>
  vi.mocked(getAuth).mockResolvedValue({
    api: { getSession: vi.fn().mockResolvedValue(user ? { user } : null) }
  } as never)

describe("proxy", () => {
  beforeEach(() => { vi.clearAllMocks() })

  it("redirects unauthenticated user from /trending to /login with callbackUrl", async () => {
    mockSession(null)
    const res = await proxy(req("/trending"))
    expect(res.status).toBe(307)
    expect(res.headers.get("location")).toContain("/login")
    expect(res.headers.get("location")).toContain("callbackUrl=%2Ftrending")
  })

  it("allows authenticated user to access /trending", async () => {
    mockSession({ id: "user_1" })
    const res = await proxy(req("/trending"))
    expect(res.status).toBe(200)
  })

  it.each(["/trending", "/analytics", "/monitor", "/idea-generator"])(
    "guards protected route %s",
    async (route) => {
      mockSession(null)
      const res = await proxy(req(route))
      expect(res.status).toBe(307)
    }
  )

  it.each(["/login", "/", "/api/auth/session"])(
    "passes through unprotected route %s",
    async (route) => {
      const res = await proxy(req(route))
      expect(res.status).toBe(200)
      expect(getAuth).not.toHaveBeenCalled()
    }
  )

  it("includes callbackUrl param when redirecting", async () => {
    mockSession(null)
    const res = await proxy(req("/analytics"))
    const location = res.headers.get("location") ?? ""
    expect(location).toContain("callbackUrl=%2Fanalytics")
  })
})
```

```typescript
// apps/dashboard/src/features/analytics/queries.test.ts
import { describe, it, expect, vi, beforeEach } from "vitest"
import { createTestDb } from "../../../tests/helpers/db"
import { posts, analyzedPosts, postTags } from "@viral-scout/database"

vi.mock("next/cache", () => ({
  cacheLife: vi.fn(),
  cacheTag: vi.fn()
}))
vi.mock("@/lib/db")

import { getDb } from "@/lib/db"
import { fetchAnalyticsData } from "./queries"

const seedPost = async (
  db: ReturnType<typeof createTestDb>["db"],
  overrides: { platform?: string; viralityScore?: number; tagId?: string } = {}
) => {
  const postId = crypto.randomUUID()
  const analyzedId = crypto.randomUUID()

  await db.insert(posts).values({
    id: postId,
    externalId: `test:${postId}`,
    platform: overrides.platform ?? "reddit",
    url: "https://example.com",
    title: "Test post",
    body: "Test body",
    authorHandle: "user",
    publishedAt: new Date().toISOString(),
    totalEngagement: 100,
    likesCount: 80,
    commentsCount: 15,
    sharesCount: 5,
    sourceId: "src_reddit_pf",
    scrapedAt: new Date().toISOString()
  })

  await db.insert(analyzedPosts).values({
    id: analyzedId,
    postId,
    viralityScore: overrides.viralityScore ?? 7.5,
    contentType: "story",
    hookAngles: "[]",
    reasoning: "test",
    analyzedAt: new Date().toISOString()
  })

  if (overrides.tagId) {
    await db.insert(postTags).values({ postId, tagId: overrides.tagId })
  }

  return { postId, analyzedId }
}

describe("fetchAnalyticsData", () => {
  let db: ReturnType<typeof createTestDb>["db"]

  beforeEach(() => {
    vi.clearAllMocks()
    const testDb = createTestDb()
    db = testDb.db
    vi.mocked(getDb).mockResolvedValue(db as never)
  })

  it("returns empty arrays when DB has no posts", async () => {
    const { platformStats, topTags, viralityDistribution } = await fetchAnalyticsData()
    expect(platformStats).toHaveLength(0)
    expect(viralityDistribution).toHaveLength(0)
  })

  it("returns platform stats for seeded posts", async () => {
    await seedPost(db, { platform: "reddit", viralityScore: 9.0 })
    await seedPost(db, { platform: "reddit", viralityScore: 7.0 })
    await seedPost(db, { platform: "youtube", viralityScore: 6.0 })

    const { platformStats } = await fetchAnalyticsData()
    const redditStat = platformStats.find((s) => s.platform === "reddit")
    expect(redditStat).toBeDefined()
    expect(redditStat?.totalPosts).toBe(2)
    expect(redditStat?.hotPosts).toBe(1) // only score 9.0 >= 8.5
  })

  it("returns top tags sorted by post count", async () => {
    // tag_01 = 'thu-nhap-thu-dong' (seeded from migration)
    await seedPost(db, { tagId: "tag_01" })
    await seedPost(db, { tagId: "tag_01" })
    await seedPost(db, { tagId: "tag_02" })

    const { topTags } = await fetchAnalyticsData()
    expect(topTags[0]?.slug).toBe("thu-nhap-thu-dong")
    expect(topTags[0]?.postCount).toBe(2)
  })

  it("returns virality distribution bucketed by rounded score", async () => {
    await seedPost(db, { viralityScore: 7.3 }) // bucket 7
    await seedPost(db, { viralityScore: 7.6 }) // bucket 8
    await seedPost(db, { viralityScore: 9.1 }) // bucket 9

    const { viralityDistribution } = await fetchAnalyticsData()
    const buckets = viralityDistribution.map((d) => d.bucket)
    expect(buckets).toContain(7)
    expect(buckets).toContain(8)
    expect(buckets).toContain(9)
  })
})
```

```typescript
// apps/dashboard/src/app/api/generate-idea/route.test.ts
import { describe, it, expect, vi, beforeEach } from "vitest"

const mockAIRun = vi.fn()
vi.mock("@opennextjs/cloudflare", () => ({
  getCloudflareContext: vi.fn().mockResolvedValue({
    env: { AI: { run: mockAIRun } }
  })
}))

import { POST } from "./route"

const validBody = {
  title: "Cách kiếm tiền online",
  body: "Tôi đã thử nhiều cách...",
  platform: "reddit",
  contentType: "story",
  hookAngles: ["Từ 0 đến 100 triệu", "Bí quyết ít ai biết"]
}

const makeRequest = (body: object) =>
  new Request("http://localhost/api/generate-idea", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(body)
  })

const makeAIStream = (text: string) =>
  new ReadableStream({
    start(controller) {
      controller.enqueue(new TextEncoder().encode(`data: {"response":"${text}"}\n\n`))
      controller.close()
    }
  })

describe("POST /api/generate-idea", () => {
  beforeEach(() => { vi.clearAllMocks() })

  it("returns 400 when title is missing", async () => {
    const res = await POST(makeRequest({ body: "test", platform: "reddit", contentType: "story", hookAngles: [] }))
    expect(res.status).toBe(400)
  })

  it("returns 400 when title is empty string", async () => {
    const res = await POST(makeRequest({ ...validBody, title: "" }))
    expect(res.status).toBe(400)
  })

  it("returns 400 when title exceeds 500 characters", async () => {
    const res = await POST(makeRequest({ ...validBody, title: "x".repeat(501) }))
    expect(res.status).toBe(400)
  })

  it("returns text/event-stream for valid request", async () => {
    mockAIRun.mockResolvedValue(makeAIStream("Hello"))
    const res = await POST(makeRequest(validBody))
    expect(res.status).toBe(200)
    expect(res.headers.get("content-type")).toContain("text/event-stream")
  })

  it("calls Workers AI with correct model", async () => {
    mockAIRun.mockResolvedValue(makeAIStream("Hello"))
    await POST(makeRequest(validBody))
    expect(mockAIRun).toHaveBeenCalledWith(
      "@cf/meta/llama-3.3-70b-instruct-fp8-fast",
      expect.objectContaining({ stream: true })
    )
  })

  it("includes title in AI prompt", async () => {
    mockAIRun.mockResolvedValue(makeAIStream("Hello"))
    await POST(makeRequest(validBody))
    const callArgs = mockAIRun.mock.calls[0]?.[1] as { messages: { content: string }[] }
    const userMessage = callArgs.messages.find((m) => m.content.includes(validBody.title))
    expect(userMessage).toBeDefined()
  })
})
```
